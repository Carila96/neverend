```javascript
let currentPriceTier = {
  monthlyFull: 1000,
  pricePerBlock: 0.40,
  totalClaimedBlocks: 0,
  nextTier: { monthlyFull: 2000, pricePerBlock: 0.70, triggerBlocks: 20001, blocksRemaining: 20001 },

async function fetchPriceTier() {
  try {
    const res = await fetch('/api/grid?tier=1');
    const data = await res.json();
    if (data.currentTier) {
      currentPriceTier = data;
      updatePriceDisplay();
    }
  } catch(e) {
    console.warn('Price tier fetch failed, using defaults');
  }
}

function updatePriceDisplay() {
  const t = currentPriceTier;
  // Update stat bar
  const statEl = document.getElementById('statPriceFrom');

  // Update full stage price
  const monoEl = document.getElementById('monoPrice');
  if (monoEl) monoEl.textContent = '$' + t.currentTier.monthlyFull.toLocaleString() + '/mo';
  
  // Update progress bar
  updateProgressBar();
  
  // Update preset prices
  updatePresetPrices();
}
```

```javascript
function updateProgressBar() {
  const t = currentPriceTier;
  if (!t.currentTier) return;
  
  document.getElementById('tierCurrentPrice').textContent = '$' + t.currentTier.monthlyFull.toLocaleString() + '/stage/mo';
  document.getElementById('tierBlocksClaimed').textContent = t.totalClaimedBlocks.toLocaleString();
  
  if (t.nextTier) {
    document.getElementById('tierNextPrice').textContent = '$' + t.nextTier.monthlyFull.toLocaleString() + '/stage/mo';
    document.getElementById('tierBlocksRemaining').textContent = t.nextTier.blocksRemaining.toLocaleString();
    const progress = ((t.totalClaimedBlocks - t.currentTier.minBlocks) / (t.nextTier.triggerBlocks - t.currentTier.minBlocks)) * 100;
    document.getElementById('tierProgressBar').style.width = Math.min(100, Math.max(0, progress)) + '%';
  } else {
    document.getElementById('tierNextPrice').textContent = 'MAX TIER';
    document.getElementById('tierBlocksRemaining').textContent = '0';
    document.getElementById('tierProgressBar').style.width = '100%';
  }
}
```

## Task 3: Update calculatePrice() to use dynamic price per block

Find the calculatePrice() function and update it to use currentPriceTier:

```javascript
function calculatePrice(blockCount) {
  const ppb = currentPriceTier.currentTier ? currentPriceTier.currentTier.pricePerBlock : 0.40;
  let base = blockCount * ppb;
  let discount = 0;
  if (blockCount >= 1000) discount = 0.30;
  else if (blockCount >= 500) discount = 0.20;
  else if (blockCount >= 200) discount = 0.10;
  const total = Math.max(1, base * (1 - discount));
  return { base, disc: discount, total, annual: total * 10 };
}
```
